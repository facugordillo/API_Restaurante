create view [dbo].[V_ReservaSucursal] 
as

select   r.Id as IdReserva
		,r.Nombre
		,s.IdSucursal
		,s.Nombre as Sucursal

from Reserva as r
inner join Sucursal as s
on r.IdSucursal = s.IdSucursal
